CREATE TABLE `imrandb`.`user_info` ( `id` INT NOT NULL AUTO_INCREMENT , `name` VARCHAR(255) NOT NULL , `email` VARCHAR(255) NOT NULL , `password` VARCHAR(40) NOT NULL , `confirm` VARCHAR(40) NOT NULL , `phone` INT(20) NOT NULL , `radio` VARCHAR(255) NOT NULL , `optradio` VARCHAR(255) NOT NULL , `dob` VARCHAR(255) NOT NULL , `image` VARCHAR(255) NOT NULL , PRIMARY KEY (`id`)) ENGINE = InnoDB;
